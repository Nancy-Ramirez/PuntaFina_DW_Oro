-- DROP TABLE IF EXISTS public.oro_tax;

CREATE TABLE IF NOT EXISTS public.oro_tax
(
    id integer NOT NULL DEFAULT nextval('oro_tax_id_seq'::regclass),
    code character varying(255) COLLATE pg_catalog."default" NOT NULL,
    description text COLLATE pg_catalog."default",
    rate double precision NOT NULL,
    created_at timestamp(0) without time zone NOT NULL,
    updated_at timestamp(0) without time zone NOT NULL,
    CONSTRAINT oro_tax_pkey PRIMARY KEY (id)
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.oro_tax
    OWNER to oro_user;

COMMENT ON COLUMN public.oro_tax.rate
    IS '(DC2Type:percent)(DC2Type:percent)';
